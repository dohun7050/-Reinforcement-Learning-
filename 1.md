## 사인파 만들기
```
import numpy as np
import matplotlib.pyplot as plt

t = np.linspace(0,1,100)
f = 1 #Hz
f2 = 2
y1 = 1*np.sin(2*np.pi*f*t + 0)
y2 = 0.7 * np.sin(2*np.pi*f*t - 1)
ySum = y1 + y2 
plt.plot(t,y1,"-",label = 'sin(2$\pi$ft)')

plt.plot(t, y2, ls="--", 
         label='0.7sin(2$\pi$ft-1)')
plt.plot(t,ySum,label = "sum",linewidth = 3)
plt.xlabel("Time")
plt.title("Sine Wave")
plt.legend()
plt.grid()
plt.show()
```
-- 주파수 바꾸기, 진폭 바꾸기, 두 소리 합치기

## 사인파 공식
y = A \sin(2\pi f t + \phi)
y=1sin(2π⋅1⋅t+0)

plt.plot(t,1*np.sin(2*np.pi*f*t + 0))

1 = 진폭(소리 크기)
t = 시간
f = 주파수(1초에 몇 번 진동)
0 = 위상(파형이 시작되는 위치)

## 개념 정리
진폭(A)이 커지면 파형의 높이가 커진다 A가 파형의 최대 높이 결정
주파수(f)가 커지면 같은 시간안에 더 많은 파가 생긴다 파형이 더 촘촘해짐

## wav 파일 생성
from scipy.io.wavfile import write

sr = 44100
t = np.linspace(0,1,sr)

y = (
np.sin(2*np.pi*440*t) +
0.5*np.sin(2*np.pi*880*t) +
0.25*np.sin(2*np.pi*1320*t)
)
write("sound2.wav",sr,y.astype(np.float32))
